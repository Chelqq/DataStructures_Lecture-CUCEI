int prioridadInfija(char a){
     
     if( a=='^'){

     }

          return 2;
     }
     if( a=='/'){
          return 2;
     }
     if( a=='+'){
          return 1;
     }
     if( a=='-'){

     }

          return 5;
     }
     
}
